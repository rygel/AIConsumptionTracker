1use std::{collections::HashMap, env::args};

4fn main() -> WMIResult<()> {
5    let wmi_con = WMIConnection::new(COMLibrary::new()?)?;
6    let args: Vec<String> = args().collect();
7    let query = match args.get(1) {
8        None => {
9            println!("Expected an argument with a WMI query");
10            return Ok(());
11        }
12        Some(query) => query,
13    };
14
15    let results: Vec<HashMap<String, Variant>> = match wmi_con.raw_query(query) {
16        Err(e) => {
17            println!("Couldn't run query {} because of {:?}", query, e);
18            return Ok(());
19        }
20        Ok(results) => results,
21    };
22
23    for (i, res) in results.iter().enumerate() {
24        println!("Result {}", i);
25        println!("{:#?}", res);
26    }
27
28    Ok(())
29}
